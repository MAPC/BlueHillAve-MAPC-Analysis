Charts

Home sale transactions (single family + condo) - annual count by price bin (200k bins) - stacked bar chart, 2000 - 2022 - BHA tracts, (pooled) - stacked bar chart using $200k bins

#data
sf_con_price <- warren_bha %>% 
  filter(restype == "R1F" | restype == "CON") %>% 
  mutate(
    price_bin = case_when(price_adj < 200000 ~ "Less than $200k",
                          price_adj >= 200000 & price_adj < 400000 ~ "$200k to <$400k",
                          price_adj >= 400000 & price_adj < 600000 ~ "$400k to <$600k",
                          price_adj >= 600000 & price_adj < 800000 ~ "$600k to <$800k",
                          price_adj >= 800000 ~ "$800k+"
                          )
  ) %>% 
  select(year, price_bin, investor_type_purchase) 
#chart
price_bin_years <- ggplot(data = sf_con_price%>%filter(year>2010), 
                     aes(x = as.factor(year), 
                         fill = factor(price_bin, levels = c("$800k+", "$600k to <$800k",
                                                             "$400k to <$600k", "$200k to <$400k", "Less than $200k"),)
                     )) +
  geom_bar() +
  scale_fill_viridis(discrete = TRUE) +

  xlab("Year") +
  ylab("Number of Sales") +

inv_purchases <- warren_mapc %>%

  group_by(ct20_id) %>% 
  mutate(ct_trans = n()) %>% 
  group_by(ct20_id, investor_type_purchase) %>%
